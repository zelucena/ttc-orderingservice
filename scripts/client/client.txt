# parar todos os containers se necessário
sudo docker container stop $(sudo docker container ls -qa)

cryptogen generate --config=./crypto-config.yaml
configtxgen -profile BFTGenesis -channelID bftchannel -outputBlock genesisblock -configPath="/bft-config/"

configtxgen -profile BFTChannel -outputCreateChannelTx channel.tx -channelID channel47 -configPath="/bft-config/"
configtxgen -profile BFTChannel -outputAnchorPeersUpdate anchor.tx -channelID channel47 -asOrg IBM -configPath="/bft-config/"

peer channel create -o 1000.frontend.bft:7050 -c channel47 -f channel.tx
peer channel update -o 1000.frontend.bft:7050 -c channel47 -f anchor.tx

# juntar ao bloco gênesis
peer channel join -b /bft-config/genesisblock
